- תצוגה מקדימה מיידית של התמונה

- עמודה "קבלה" בטבלת הוצאות

- לחיצה על 📷 → מודל עם הקבלה במסך מלא

#### קובץ: `js/app.js`
```javascript
// פונקציה חדשה לניווט מהיר
function navigateToPayments(tenantId) {
    // 1. מעבר ללשונית תשלומים
    switchTab('payments');
    
    // 2. הגדרת סינון לפי דייר
    const filterSelect = document.getElementById('paymentTenantFilter');
    if (filterSelect) {
        filterSelect.value = tenantId;
    }
    
    // 3. סינון תשלומים
    renderPaymentsTable(tenantId);
    
    // 4. גלילה לראש העמוד
    window.scrollTo({ top: 0, behavior: 'smooth' });
    
    // 5. הצגת הודעה
    const tenant = appState.tenants.find(t => t.id === tenantId);
    if (tenant) {
        showToast(`מציג תשלומים של ${tenant.name}`, 'info');
    }
}
```
